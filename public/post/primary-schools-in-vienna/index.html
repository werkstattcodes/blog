library(here)
shared_df_main <- readr::read_csv2(here("blog_data", "Vienna_primary_schools",
                                        "shared_df_main.csv"))
library(crosstalk)
shared_df_main <- SharedData$new(shared_df_main)



# > scatter plotly --------------------------------------------------------
library(plotly)

plot_distance_diff <-  
  plot_ly(
    data=shared_df_main,
    mode='markers',
    type="scatter"
    ) %>% 
  add_trace(
    text=~tooltip,
    marker=list(color='green'),
    x=~distance,
    y=~diff_factor,
    showlegend=F
  )
    
plot_distance_diff

cross_map2 <- plot_mapbox(

    y = ~school_2_lat,
    x = ~school_2_long,
    color = "#FF0000",
    text = ~address2
  ) %>%

    mapbox = list(style = "mapbox://styles/zoowalk/ck6kz867d0jdi1io2lxyp9008",
                  zoom = 10,

                  ),

    #margin=0
    )

# > combine graphs -----------------------------------------------------------

  cross_map2)
